class Solution
{
public:
    int compare2_0(string newest, vector<string> previous)
    {
        string biggest = previous[0];
        for (int i = 1; i < previous.size(); i++)
        {
            int result = Compare(biggest, previous[i]);
            if (result == -1)
                biggest = previous[i];
        }
        int ans = Compare(newest, biggest);
        return ans;
    }
};
